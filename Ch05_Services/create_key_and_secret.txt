# generate 2048 bit private key and save to file
openssl genrsa -out tls.key 2048

# generate self-signed certificate using the private key for the host
openssl req -new -x509 -key tls.key -out tls.cert -days 360 -subj /CN=kubia.example.com

# create kubernetes secret using the self-signed cert and private key
kubectl create secret tls tls-secret --cert=tls.cert --key=tls.key
